# SCOURGE VIRUS EVENT CHAIN
# Narrative event series that kills most of the Viltrumite population at game start
# Reduces population from thousands to ~100 survivors over time
# Triggered by on_game_start_country and on_yearly_pulse_country

namespace = viltrum_start_pops

# Event 1: Scourge Virus Outbreak Begins
# Initial event that starts the population culling chain
# Only fires once per empire via country flag
country_event = {
    id = viltrum_start_pops.1
    title = "viltrum_start_pops.1.name"
    desc = "viltrum_start_pops.1.desc"
    picture = "GFX_evt_sv_0"  # Custom event picture (sv-0.dds)
    show_sound = event_bad_omen
    is_triggered_only = yes

    trigger = {
        # Only for Viltrum Survivors origin
        has_origin = origin_viltrum_survivors
        
        # Prevent event from firing multiple times
        NOT = { has_country_flag = viltrum_scourge_started }
    }

    immediate = {
        # Mark virus as started
        set_country_flag = viltrum_scourge_started
    }

    option = {
        name = "viltrum_start_pops.1.a"
        hidden_effect = {
            # Start the population culling loop on capital planet
            capital_scope = {
                planet_event = { id = viltrum_start_pops.2 days = 15 }
            }
        }
    }
}

# Event 2: Scourge Virus Pop Culling Loop (Hidden)
# Repeating event that kills pops every 10 days
# Shows progress events at specific population milestones
# Stops when population drops below 100
planet_event = {
    id = viltrum_start_pops.2
    hide_window = yes  # No popup - runs silently
    is_triggered_only = yes

    trigger = {
        owner = {
            # Virus must be active
            has_country_flag = viltrum_scourge_started
            # And not yet ended
            NOT = { has_country_flag = viltrum_scourge_ended }
        }
    }

    immediate = {
        # Kill 100 population (1 pop = 100 individuals)
        random_owned_pop_group = {
            limit = {
                pop_group_size > 1
            }
            kill_single_pop = yes
        }
        
        # PROGRESS EVENT 1: First Wave (at ~3000-3500 pops)
        if = {
            limit = {
                pop_amount > 3000
                pop_amount < 3500
                owner = {
                    NOT = { has_country_flag = viltrum_progress_1 }
                }
            }
            owner = {
                set_country_flag = viltrum_progress_1
                country_event = { id = viltrum_start_pops.4 }
            }
        }
        
        # PROGRESS EVENT 2: The Culling (at ~1500-2000 pops)
        else_if = {
            limit = {
                pop_amount > 1500
                pop_amount < 2000
                owner = {
                    NOT = { has_country_flag = viltrum_progress_2 }
                }
            }
            owner = {
                set_country_flag = viltrum_progress_2
                country_event = { id = viltrum_start_pops.5 }
            }
        }
        
        # PROGRESS EVENT 3: Final Days (at ~600-800 pops)
        else_if = {
            limit = {
                pop_amount > 600
                pop_amount < 800
                owner = {
                    NOT = { has_country_flag = viltrum_progress_3 }
                }
            }
            owner = {
                set_country_flag = viltrum_progress_3
                country_event = { id = viltrum_start_pops.6 }
            }
        }
        
        # CHECK STOPPING CONDITION
        if = {
            limit = {
                # Continue if above 100 pops
                pop_amount > 100
            }
            # Schedule next culling cycle in 10 days
            planet_event = { id = viltrum_start_pops.2 days = 10 }
        }
        else = {
            # Population reached target - end virus
            owner = {
                country_event = { id = viltrum_start_pops.3 }
            }
        }
    }
}
